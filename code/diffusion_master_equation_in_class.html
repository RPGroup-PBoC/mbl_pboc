# first loop through time
for t in range(1, n_steps):
    
    # then loop through all the interior boxes
    for i in range(1, n_boxes-1):
        p[t, i] = 0.5*p[t-1, i-1] + 0.5*p[t-1, i+1]
        
    # special case for the left-most box
    p[t, 0] = p[t-1, 0] - 0.5*p[t-1, 0] + 0.5*p[t-1, 1]
    
    # special case for the right-most box
    p[t, n_boxes-1] = p[t-1, n_boxes-1] - 0.5*p[t-1, n_boxes-1] + 0.5*p[t-1, n_boxes-2]
