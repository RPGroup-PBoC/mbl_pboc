# simulation parameters
n_steps = 1000
n_walkers = 1000

# initialize array for storing walker trajecctories
positions_2D = np.zeros([n_walkers, n_steps])

# first loop through the walkers
for i in range(n_walkers):
    
    # then loop through steps
    for j in range(1, n_steps):
        
        # generate random number
        rand = np.random.uniform()
        
        if rand > 0.5:
            positions_2D[i,j] = positions_2D[i,j-1] + 1 
            
        else:
            positions_2D[i,j] = positions_2D[i,j-1] - 1

means = np.zeros(n_steps)

# loop through steps to get mean position at each time
for i in range(n_steps):
    mean = np.mean(positions_2D[:,i])
    means[i] = mean
